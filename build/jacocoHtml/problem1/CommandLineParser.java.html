package problem1;

import java.util.HashMap;
import java.util.Map;
import java.util.Set;
import java.util.regex.Pattern;

public class CommandLineParser {

  private final String[] args;
  private boolean ifLetter = false;
  private boolean ifEmail = false;

  // Legal commands set
  private static final Set<String> LEGAL_COMMANDS = Set.of(
      "--email-template", "--letter-template",
      "--output-dir", "--csv-file",
      "--letter", "--email"
  );

  // constructor
  public CommandLineParser(String[] args) throws IllegalCommandException {
//    if (!validateArguments(args)) {
//      throw new IllegalArgumentException("Invalid command line arguments.");
//    }
    this.args = args;
  }


  public boolean validateArguments(String[] args) throws IllegalCommandException {
    // Check if each argument follows the expected format
    for (String arg : args) {
      if (!isValidFormat(arg) || !checkIllegalCommand(arg)) {
        return false;
      }
    }
    return true;
  }

  boolean isValidFormat(String value) {
    // Expression pattern for file paths
    String pathPattern = "([\\w.]+/(\\w+/)*)";
    // Expression pattern for file extensions
    String extensionPattern = "\\.((csv)|(txt))$";
    return Pattern.matches(pathPattern, value) && Pattern.matches(extensionPattern, value);
  }

  private boolean checkIllegalCommand(String arg) throws IllegalCommandException {
      if (!LEGAL_COMMANDS.contains(arg) ){
        throw new IllegalCommandException(arg + " is not a legal command.");
      }
    return true;
  }


  public Map<String, String> parseArguments() {
    Map<String, String> arguments = new HashMap<>();
    boolean emailTemplatePresent = false;
    boolean letterTemplatePresent = false;
    boolean outputDirPresent = false;
    boolean csvFilePresent = false;


    for (int i = 0; i < args.length; i++) {
      if (args[i].startsWith("--")) {
        String key = args[i].substring(2); // ignore --
        String value = null;

        // change the flags if possible
        if (key.equals("email-template")) {
          emailTemplatePresent = true;
        } else if (key.equals("letter-template")) {
          letterTemplatePresent = true;
        } else if (key.equals("output-dir")) {
          outputDirPresent = true;
        } else if (key.equals("csv-file")) {
          csvFilePresent = true;
        } else if (key.equals("letter")) {
          ifLetter = true;
          continue;
        } else if (key.equals("email")) {
          ifEmail = true;
          continue;
        }
        // extract the value
        if (i < args.length - 1 && !args[i + 1].startsWith("-")) {
            value = args[i + 1];
            i++;
        }
        arguments.put(key, value);
        }
      }


    // Check if the required options are present together
    if (ifEmail && !emailTemplatePresent) {
      //todo exception class
      System.err.println("Error: --email provided but no --email-template was given.");
      return null;
    }

    if (ifLetter && !letterTemplatePresent) {
      //todo exception class
      System.err.println("Error: --letter provided but no --letter-template was given.");
      return null;
    }

    if (!outputDirPresent) {
      //todo exception class
      System.err.println("Error: --output-dir is required.");
      return null;
    }

    if (!csvFilePresent) {
      //todo exception class
      System.err.println("Error: --csv-file is required.");
      return null;
    }
    return arguments;
  }

  public boolean isIfLetter() {
    return ifLetter;
  }

  public boolean isIfEmail() {
    return ifEmail;
  }

  public static void main(String[] args) {
      try {
        CommandLineParser parser = new CommandLineParser(args);
        Map<String, String> arguments = parser.parseArguments();
        printArguments(arguments, parser);
      } catch (IllegalCommandException e) {
        System.err.println("Error: " + e.getMessage());
      }
  }


  private static void printArguments(Map<String, String> arguments, CommandLineParser parser) {
    System.out.println("Parsed Arguments:");
    for (Map.Entry<String, String> entry : arguments.entrySet()) {
      System.out.println(entry.getKey() + ": " + entry.getValue());
    }
    System.out.println("ifLetter: " + parser.isIfLetter());
    System.out.println("ifEmail: " + parser.isIfEmail());
  }
}
