package problem1;

import java.io.*;
import java.util.*;

public class LetterTemplateGenerator {
  public void generateLetters(Map<String, String> arguments, List<Map<String, String>> csvData, String outputDir) {
    String letterTemplatePath = arguments.get("--letter-template");
    String letterContent = readTemplate(letterTemplatePath);

    if (letterContent == null) {
      System.err.println("Error reading letter template.");
      return;
    }

    for (Map<String, String> rowData : csvData) {
      String generatedLetter = generateLetter(letterContent, rowData);
      String outputFile = outputDir + File.separator + "letter_" + rowData.get("email") + ".txt";
      writeOutput(generatedLetter, outputFile);
    }
  }

  private String readTemplate(String filePath) {
    StringBuilder content = new StringBuilder();
    try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
      String line;
      while ((line = reader.readLine()) != null) {
        content.append(line).append("\n");
      }
    } catch (IOException e) {
      e.printStackTrace();
      return null;
    }
    return content.toString();
  }

  private String generateLetter(String letterTemplate, Map<String, String> data) {
    String generatedLetter = letterTemplate;
    for (Map.Entry<String, String> entry : data.entrySet()) {
      String placeholder = "[[" + entry.getKey() + "]]";
      generatedLetter = generatedLetter.replace(placeholder, entry.getValue());
    }
    return generatedLetter;
  }

  private void writeOutput(String content, String filePath) {
    try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath))) {
      writer.write(content);
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
