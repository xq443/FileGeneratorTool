package problem1;

import java.io.*;
import java.util.*;

public class EmailTemplateGenerator {

  public void generateEmails(Map<String, String> arguments, List<Map<String, String>> csvData, String outputDir) {
    String emailTemplatePath = arguments.get("--email-template");
    String emailSubject = "Subject: Insurance company â€“ information about recent data breach\n";
    String emailContent = readTemplate(emailTemplatePath);

    if (emailContent == null) {
      System.err.println("Error reading email template.");
      return;
    }

    for (Map<String, String> rowData : csvData) {
      String generatedEmail = generateEmail(emailContent, emailSubject, rowData);
      String outputFile = outputDir + File.separator + "email_" + rowData.get("email") + ".txt";
      writeOutput(generatedEmail, outputFile);
    }
  }

  private String readTemplate(String filePath) {
    StringBuilder content = new StringBuilder();
    try (BufferedReader reader = new BufferedReader(new FileReader(filePath))) {
      String line;
      while ((line = reader.readLine()) != null) {
        content.append(line).append("\n");
      }
    } catch (IOException e) {
      e.printStackTrace();
      return null;
    }
    return content.toString();
  }

  private String generateEmail(String emailTemplate, String emailSubject, Map<String, String> data) {
    String generatedEmail = emailSubject + emailTemplate;
    for (Map.Entry<String, String> entry : data.entrySet()) {
      String placeholder = "[[" + entry.getKey() + "]]";
      generatedEmail = generatedEmail.replace(placeholder, entry.getValue());
    }
    return generatedEmail;
  }

  private void writeOutput(String content, String filePath) {
    try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath))) {
      writer.write(content);
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
